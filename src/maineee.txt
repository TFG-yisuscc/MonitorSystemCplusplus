#include "llama.h"
#include <cstdio>
#include <string>
#include <vector>

int main() {
    // 1. Inicializar el backend (CPU/GPU)
    llama_backend_init();

    // 2. Parámetros del modelo
    llama_model_params model_params = llama_model_default_params();
    model_params.n_gpu_layers = 0; // capas en GPU (0 = solo CPU)

    // 3. Cargar el modelo
    llama_model* model = llama_model_load_from_file(
        "/home/yisus/Descargas/tinyllama-1.1b-chat-v1.0.Q4_0.gguf",
        model_params
    );
    if (!model) {
        fprintf(stderr, "Error cargando modelo\n");
        return 1;
    }

    // 4. Crear contexto de inferencia
    llama_context_params ctx_params = llama_context_default_params();
    ctx_params.n_ctx    = 4096;  // tamaño del contexto
    ctx_params.n_batch  = 512;   // batch de procesamiento
    ctx_params.no_perf  = false;
    llama_context* ctx = llama_init_from_model(model, ctx_params);

    // Obtener vocabulario
    const llama_vocab* vocab = llama_model_get_vocab(model);

    // 5. Tokenizar el prompt
    std::string prompt = "¿Cuál es la capital de Francia?";

    std::vector<llama_token> tokens(prompt.size() + 10);
    int n_tokens = llama_tokenize(
        vocab,
        prompt.c_str(),
        prompt.size(),
        tokens.data(),
        tokens.size(),
        true,
        false
    );
    tokens.resize(n_tokens);

    // 6. Crear batch y evaluar prompt
    llama_perf_context_reset(ctx);
    llama_batch batch = llama_batch_get_one(tokens.data(), tokens.size());
    llama_decode(ctx, batch);

    // 7. Bucle de generación (sampling)
    llama_sampler* sampler = llama_sampler_chain_init(
        llama_sampler_chain_default_params()
    );
    llama_sampler_chain_add(sampler, llama_sampler_init_temp(0.7f));
    llama_sampler_chain_add(sampler, llama_sampler_init_dist(42));

    int n_cur = tokens.size();
    int n_max = 512; // máximo de tokens a generar

    while (n_cur < n_max) {
        llama_token token_id = llama_sampler_sample(sampler, ctx, -1);

        // Verificar fin de secuencia
        if (llama_vocab_is_eog(vocab, token_id)) break;

        // Convertir token a texto e imprimir
        char buf[256];
        int n = llama_token_to_piece(vocab, token_id, buf, sizeof(buf), 0, false);
        if (n > 0) {
            buf[n] = '\0';
            printf("%s", buf);
            fflush(stdout);
        }

        // Evaluar el token generado
        batch = llama_batch_get_one(&token_id, 1);
        if (llama_decode(ctx, batch) != 0) break;

        n_cur++;
    }
    printf("\n");

    // 8. Liberar recursos

    llama_perf_context_print(ctx);
    llama_perf_sampler_print(sampler);
    llama_sampler_free(sampler);
    llama_free(ctx);
    llama_model_free(model);
    llama_backend_free();

    return 0;
}
